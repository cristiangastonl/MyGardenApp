    const soilCheckSilentPlants = useMemo(() => {
      const lat = location?.lat ?? null;
      const silent = plants.filter((plant) => {

        return !isSameDay(nextCheckIn, today);

      return silent.sort((a, b) => a.name.localeCompare(b.name));
