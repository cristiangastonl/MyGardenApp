    /**
     * Outdoor category typeIds (Plant.typeId / PlantDBEntry.category). Locked in

      const ns: 'indoor' | 'outdoor' = OUTDOOR_TYPE_IDS.has(input.typeId) ? 'outdoor' : 'indoor';
